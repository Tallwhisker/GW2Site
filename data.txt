function groceries(array) {
  let temp = [];
  console.log(`ArrayLength: ${array.length}`);

  if (array.length < 2) {
    array.forEach(word => temp.push(word.item));
    return temp.join('')

  } else if (array.length === 2) {
    array.forEach(word => temp.push(word.item));
    temp.splice(-1, 0, ' and ');
    return temp.join('');

  } else {
    array.forEach(word => {
      temp.push(word.item);
      temp.push(', ');
    })
      temp.pop();
      temp.splice(-2, 1, ' and ');
      return temp.join('');
  }
  return temp;
}
